    \begin{axis}[

        axis lines=none,

        xmin =  0,

        domain = 0:8.8,

        \addplot+[name path=A,ultra thick, black] { 0.5*sign(sin(deg(x*1.75)))+1.4};

        \addplot[gray, opacity=0.5] fill between[of=A and B];
    \end{axis}
