\def\simplecuboid#1#2#3#4#5#6#7{ % #1=t #2=f #3=P #4=color #5=deltaT #6=deltaF #7=deltaP
    \fill[#4!70!white] ( 0+#5,#2+#6, #7) -- (#1+#5,#2+#6, #7) -- (#1+#5,#2+#6,#3+#7) -- ( 0+#5,#2+#6,#3+#7) -- cycle;
    \fill[#4]          ( 0+#5,#2+#6,#3+#7) -- (#1+#5,#2+#6,#3+#7) -- (#1+#5, #6,#3+#7) -- ( 0+#5, #6,#3+#7) -- cycle;
    \fill[#4!60!white] (#1+#5,#2+#6,#3+#7) -- (#1+#5,#2+#6, #7) -- (#1+#5, #6, #7) -- (#1+#5, #6,#3+#7) -- cycle;
}
    
\def\simpleaxes#1#2#3{
    \draw[-Triangle] (0,0,0) -- (#1,0,0) node[anchor=west]{$t$};
    \draw[-Triangle] (0,0,0) -- (0,#2,0) node[anchor=south]{$f$};
    \draw[-Triangle] (0,0,0) -- (0,0,#3) node[anchor=north east]{$P$};
}

\def\dashedaxes#1#2#3{
    \draw[dashed] (0,0,0) -- (#1,0,0);
    \draw[dashed] (0,0,0) -- (0,#2,0);
    \draw[dashed] (0,0,0) -- (0,0,#3);
}
