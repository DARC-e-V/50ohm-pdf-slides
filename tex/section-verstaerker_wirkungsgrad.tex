\begin{frame}
\frametitle{Lösungsweg}
\begin{itemize}
  \item gegeben: $U = 12,5V$
  \item gegeben: $I = 16A$
  \item gegeben: $P_{ab} = 90W$
  \item gesucht: $\eta$
  \end{itemize}
    \pause
    $P_{zu} = U \cdot I = 12,5V \cdot 16A = 200W$
    \pause
    $\eta = \frac{P_{ab}}{P_{zu}} = \frac{90W}{200W} = 45\%$



\end{frame}
