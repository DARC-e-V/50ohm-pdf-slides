\frametitle{Lösungweg}
\begin{itemize}
  \item gegeben: $U_0 = 0,6V$
  \item gegeben: $I_k = 1A$
  \item gegeben: $N_R = 30, N_P = 4$
  \item gesucht: $U_{0,ges}, I_{k,ges}$
  \end{itemize}
    \pause
    $U_{0,ges} = N_R \cdot U_0 = 30 \cdot 0,6V = 18V$
    \pause
    $I_{0,ges} = N_P \cdot I_k = 4 \cdot 1A = 4A$
